# Note this doesn't install the icon, that's done later

if(EMSCRIPTEN)
    # Emscripten ports are linked via flags, not CMake targets
    target_link_options(abuse PRIVATE
        "SHELL:-sUSE_SDL=2"
        "SHELL:-sUSE_SDL_MIXER=2"
        "SHELL:-sSDL2_MIXER_FORMATS=['ogg','mod','wav']"
    )
else()
    # Native or vcpkg builds link to actual imported targets
    target_link_libraries(abuse
        PRIVATE
        SDL2::SDL2
        SDL2_mixer::SDL2_mixer
    )
endif()

    target_link_libraries(abuse ${OPENGL_LIBRARIES})
